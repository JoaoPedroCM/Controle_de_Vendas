CREATE TABLE CLIENTE(
	ID_CLIENTE INT AUTO_INCREMENT PRIMARY KEY,
    NOME_CLIENTE CHAR(40) NOT NULL,
    TELEFONE_CLIENTE CHAR(13),
    EMAIL_CLIENTE CHAR(40),
    ENDERECO_CLIENTE CHAR(60)
);
CREATE TABLE VENDA(
	ID_VENDA INT AUTO_INCREMENT PRIMARY KEY,
    CLIENTE_VENDA INT NOT NULL,
    FOREIGN KEY (CLIENTE_VENDA) REFERENCES CLIENTE(ID_CLIENTE),
    DESCRICAO_VENDA CHAR(40) NOT NULL,
    VALOR_VENDA DECIMAL(6,2) NOT NULL,
    DATA_VENDA DATE NOT NULL,
    PAGO INT(1) NOT NULL
);
